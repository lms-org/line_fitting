1function [X, Y, PHI] = getPointsFromState(r, delta)
 2
 3P = zeros(numel(r), 3); %x, y, phi
 4phi = r(2);
 5
 6%erster Punkt 
 7P(1, 1) = 0;
 8P(1, 2) = r(1);
 9P(1, 3) = phi;
10
11%zweiter Punkt
12P(2, 1) = P(1, 1) + delta*cos(r(2));
13P(2, 2) = P(1, 2) + delta*sin(r(2));  
14P(2, 3) = phi;
15    
16for s=3:numel(r)   
17        phi = phi + pi - 2*acos(delta*r(s)/2); 
18        P(s, 1) = P(s-1, 1) + delta*cos(phi);
19        P(s, 2) = P(s-1, 2) + delta*sin(phi);
20        P(s, 3) = phi;  
21end  
22
23X = P(:, 1);
24Y = P(:, 2);
25PHI = P(:, 3);
