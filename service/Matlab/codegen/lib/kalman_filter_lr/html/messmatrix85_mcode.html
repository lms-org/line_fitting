1function H = messmatrix(P, r, delta, ind, lambda)
 2
 3num_points = numel(r);
 4d = delta;
 5
 6%Ableitungen nach phi0
 7d_phi0 = zeros(num_points, 2);
 8for p=2:num_points
 9    d_phi0(p, 1) = d_phi0(p-1, 1) - d*sin(P(p, 3)); %x-Komponenten
10    d_phi0(p, 2) = d_phi0(p-1, 2) + d*cos(P(p, 3)); %y-Komponenten
11end
12
13%Ableitungen nach den Kr√ºmmungen c_i
14d_c_x = zeros(num_points, num_points); %x-Komponenten
15d_c_y = zeros(num_points, num_points); %y-Komponenten
16for p=3:num_points
17    for c=2:num_points-1
18        if p>c     
19            d_c_x(p, c) = d_c_x(p-1, c) - d^2*sin(P(p, 3))/sqrt(1-d^2*r(c+1)^2/4);
20            d_c_y(p, c) = d_c_y(p-1, c) + d^2*cos(P(p, 3))/sqrt(1-d^2*r(c+1)^2/4);
21        end        
22    end   
23end
24
25
26%Messmatrix zusammensetzen
27H = zeros(2*numel(lambda), numel(r));
28count = 1;
29for i=1:numel(ind) %alle Messpunkte iterieren
30    H(count, 1)   = 0;   %Ableitung nach y0
31    H(count+1, 1) = 1;   
32    
33    H(count, 2)   = d_phi0(ind(i), 1);   %Ableitung nach phi0
34    H(count+1, 2) = d_phi0(ind(i), 2);   
35    
36    for c=3:ind(i) %Ableitung nach Kr√ºmmungen c 
37        if c==ind(i)
38            H(count, c)   = d_c_x(ind(i), c-1) + lambda(i)*d^2*sin(P(ind(i)+1, 3))/sqrt(1-d^2*r(c-1)^2/4);
39            H(count+1, c) = d_c_y(ind(i), c-1) - lambda(i)*d^2*cos(P(ind(i)+1, 3))/sqrt(1-d^2*r(c-1)^2/4);
40        else
41            H(count, c)   = d_c_x(ind(i), c-1);
42            H(count+1, c) = d_c_y(ind(i), c-1);
43        end           
44    end
45    
46    count = count + 2;
47end
48
49end
