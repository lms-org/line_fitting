1function [r, Pk] = kalman_filter_lr (r, delta_x, delta_y, delta_phi, Pk, Q, R_fakt, delta, xl, yl, xr, yr, xm, ym, interp_mode, prior_fact) %#codegen
  2
  3% "kalman_filter_lr" 
  4% 
  5% ----Nomenklatur----
  6%  * r: Zustandsvektor = [y0, phi0, kappa_1, kappa_2, ... , kappa_(n-1)] 
  7%   --> y0: y-Wert des ersten Punktes (x-Wert ist immer 0)
  8%   --> phi0: Anfangssteigung zwischen erstem und zweitem Punkt
  9%   --> kappa_i: Krümmungen an den entsprechenden Punkten (Krümmungsdefinition: 1/R, 
 10%       wobei R der Radius des Kreises ist, auf dem der betrachtete Punkt, sein Vorgänger 
 11%       und sein Nachfolger liegen)
 12%  * delta_x, delta_y, delta_phi: skalare Werte für die Eigenbewegung des Fahrzeugs seit dem letzten Aufruf des Filters (im KOSY des letzten Aufrufs); 
 13%    ist delta_x = delta_y = delta_phi = 0, wird die Berechnung einer state-transition Matrix deaktiviert
 14%  * Pk: Kovarianzmatrix des Zustands dim[n x n], mit n = Dimension des Zustandsvektors
 15%  * Q: Kovarianzmatrix des Zustandsübergangs (Prozessrauschen) dim[n x n] (symmetrische Matrix, mit weg von der 
 16%       Diagonalen abnehmenden Eintraegen -> wie stark hängen die Krümmungen an den verschiedenen Punkten zusammen?)
 17%  * R_fakt: Varianz der Messwerte (Messrauschen)
 18%  * delta: Abstand zwischen aufeinanderfolgenden Punkten (delta*n ergibt die Länge des praedizierten Fahrstreifens)
 19%  * xl, yl: Vektoren mit den Messwerten für die linke Spur
 20%  * xr, yr: Vektoren mit den Messwerten für die rechte Spur
 21%  * xm, ym: Vektoren mit den Messwerten für die Mittellinie
 22%  * interp_mode: 0 für lineare Interpolation der Krümmungen bei state transition, 1 für quadratische Interpolation
 23% 
 24%  ----Grober Ablauf des Algorithmus----
 25%  1. Projektion der Punkte von der Mittellinie nach links bzw. rechts
 26%  2. Fuer jeden Messpunkt: Berechnung des kleinsten Abstands zum aktuell praedizierten Strassenverlauf
 27%  3. Assemblierung der Jakobimatrix fuer die Projektion aus dem Zustandsraum von r auf x-y-Koordinaten
 28%  4. Kalman-Filter anwenden
 29%  5. Zustandsbegrenzungen einbringen (mathematisch unschön aber praktisch funktionierts)
 30
 31
 32%% Messpunkte verarbeiten
 33
 34% linke Seitenlinie
 35[Hl, zl, zml, nPointsL] = handle_measurements(r, delta, 0.38, xl, yl);
 36
 37% rechte Seitenlinie
 38[Hr, zr, zmr, nPointsR] = handle_measurements(r, delta, -0.38, xr, yr);
 39
 40% Mittellinie
 41[Hm, zm, zmm, nPointsM] = handle_measurements(r, delta, 0, xm, ym);
 42
 43% Linien in eine Matrix kombinieren
 44H = [Hl; Hr; Hm]; %Messmatrix
 45z = [zl; zr; zm]; %Messwerte
 46zm = [zml; zmr; zmm]; %erwartete Werte
 47
 48
 49%% KALMAN FILTER    
 50
 51% state transition
 52if delta_x == 0 && delta_y == 0 && delta_phi == 0
 53    A = eye(numel(r));
 54else
 55    [r, A] = state_transition(r, delta, delta_x, delta_y, delta_phi, interp_mode);    
 56end
 57
 58R = R_fakt.*eye(2*(nPointsL + nPointsR + nPointsM)); 
 59
 60% Filtergleichungen EKF
 61Pk = A*Pk*A' + Q;
 62y_tilde = z - zm;
 63K = Pk*H'/(H*Pk*H' + R);
 64r = r + K*y_tilde;
 65Pk = (eye(numel(r)) - K*H)*Pk;
 66
 67% Lukas: zusätzlicher prior 
 68if prior_fact > 0
 69    HH = inv(Pk + prior_fact*Q);
 70    Pk = Pk*HH*(prior_fact*Q);
 71    r = (prior_fact*Q)*HH*r;
 72end
 73
 74
 75
 76%% Zustandsbegrenzungen
 77% Krümmungen
 78kr = 0.75; % bezogen auf Mittellinie, 0.72 entspricht einem minimalen Innenradius von 1m
 79for i=3:numel(r)
 80    r(i) = min(kr, max(-kr, r(i)));
 81end
 82
 83% y-Wert des ersten Punktes
 84y_max = 0.5;
 85r(1) =  max(-y_max, min(y_max, r(1)));
 86
 87% Startwinkel 
 88phi_max = 0.79;
 89r(2) =  max(-phi_max, min(phi_max, r(2))); % 0.79 rad = 45 Grad
 90
 91
 92
 93
 94
 95
